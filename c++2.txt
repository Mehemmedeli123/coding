using namespace std;
#include <math.h>

{
    float a,b,t,y;
    cin>>a>>b>>t;
    if(t<2){
        if(t>0){
            y=a*pow(t,2)*log(t);
            }
        else{
            cout<<"t must be bigger than 0";
        }
    }
    else if(t > 2){
        y=a*cos(b*t);
        /*when you check this on calculator dont forget that this is radian not degree :)*/
    }
    else{
        y=1;
    }
    cout<<"y="<<y;

}
